%Define reference to an arbitrary number of equations (\eqs{label_1,label_2....,label_n} will show eqs. ref_1, ref_2, ..., and ref_n)
\newcommand{\eqs}[1]{\setcounter{eqsNumArgs}{0}\foreach\i in{#1}{\stepcounter{eqsNumArgs}}%
	\ifthenelse{\equal{\theeqsNumArgs}{1}}{eq.~(\ref{#1})}%
	{\ifthenelse{\equal{\theeqsNumArgs}{2}}%
		{eqs.~\foreach\i[count=\q]in{#1}{\ifthenelse{\equal{\q}{\theeqsNumArgs}}{and (\ref{\i})}{(\ref{\i})~}}}%
		{eqs.~\foreach\i[count=\q]in{#1}{\ifthenelse{\equal{\q}{\theeqsNumArgs}}{and (\ref{\i})}{(\ref{\i}),~}}}}}


%Define reference to an arbitrary number of equations (\Eqs{label_1,label_2....,label_n} will show Eqs. ref_1, ref_2, ..., and ref_n)

		{Eqs.~\foreach\i[count=\q]in{#1}{\ifthenelse{\equal{\q}{\theEqsNumArgs}}{and (\ref{\i})}{(\ref{\i}),~}}}}}


%Define reference to an arbitrary number of labels (\REF{label_1,label_2....,label_n} will show ref_1, ref_2, ..., and ref_n)
